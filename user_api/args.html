 1int main(int argc, char ** argv) {
 2    Args args;
 3    ArgToolchain arg_gcc_toolchain;
 4    args.AddToolchain("gcc", "Generic GCC toolchain", arg_gcc_toolchain);
 5
 6    // TODO, Add ArgTarget example (Currently incomplete)
 7    args.Parse(argc, argv);
 8
 9    // Root
10    args.GetProjectRootDir(); // Contains ``root_dir`` value
11    args.GetProjectBuildDir(); // Contains ``build_dir`` value
12    args.GetLogLevel(); // Contains ``loglevel`` enum
13    args.Clean(); // Contains ``clean`` value
14
15    // Toolchain
16    // .build, .test
17    arg_gcc_toolchain.state;
18    // .id, .name, .asm_compiler, .c_compiler, .cpp_compiler, .archiver, .linker -> BaseToolchain
19    BaseToolchain gcc_toolchain = arg_gcc_toolchain.ConstructToolchain();
20
21    // Underlying CLI11 library
22    auto & app = args.Ref();
23    const auto & app = args.ConstRef();
24
25    return 0;
26}
